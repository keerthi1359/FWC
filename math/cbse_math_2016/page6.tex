\begin{enumerate}[leftmargin=*, label=\arabic*., start=10, itemsep=16mm]

\item
\textHindi{%
$x^{\sin x} + (\sin x)^{\cos x}$ का $x$ के सापेक्ष अवकलन कीजिए।
}\\[4mm]
\vspace{6mm}

\begin{center}
\textHindi{\textbf{अथवा}}\\[2mm]
\end{center}

\vspace{6mm}
\textHindi{%
यदि $y = 2\cos(\log x) + 3\sin(\log x)$ है, तो सिद्ध कीजिए कि
\[
x^2 \frac{d^2 y}{dx^2} + x \frac{dy}{dx} + y = 0.
\]
}\\[4mm]




Differentiate $x^{\sin x} + (\sin x)^{\cos x}$ with respect to $x$.
\vspace{6mm}

\begin{center}
\textbf{OR}
\end{center}

If $y = 2\cos(\log x) + 3\sin(\log x)$, prove that
\[
x^2 \frac{d^2 y}{dx^2} + x \frac{dy}{dx} + y = 0.
\]

\item
\textHindi{%
यदि $x = a \sin 2t (1+\cos 2t)$ तथा
$y = b \cos 2t (1-\cos 2t)$ है, तो
$t=\dfrac{\pi}{4}$ पर $\dfrac{dy}{dx}$ ज्ञात कीजिए।
}\\[4mm]
If $x = a \sin 2t (1+\cos 2t)$ and
$y = b \cos 2t (1-\cos 2t)$, find $\dfrac{dy}{dx}$ at
$t=\dfrac{\pi}{4}$.

\item
\textHindi{%
यदि वक्र $y^2 = ax^3 + b$ के बिंदु $(2,3)$ पर
स्पर्श रेखा का समीकरण $y = 4x - 5$ है,
तो $a$ तथा $b$ के मान ज्ञात कीजिए।
}\\[4mm]
The equation of tangent at $(2,3)$ on the curve
$y^2 = ax^3 + b$ is $y = 4x - 5$.
Find the values of $a$ and $b$.

\item
\textHindi{%
ज्ञात कीजिए :
\[
\int \frac{x^2}{x^4 + x^2 - 2}\,dx
\]
}\\[4mm]
Find :
\[
\int \frac{x^2}{x^4 + x^2 - 2}\,dx
\]

\end{enumerate}
